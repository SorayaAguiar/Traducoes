function	script	F_AuctionWarper	{
	mes "[Guia da Sala de Leilão]";
	mes "Olá, você gostaria";
	mes "de entrar na";
	mes "Sala de Leilão?";
	next;
	if (select("- Sim:- Não") == 1) {
		if (getarg(0) == 1 || getarg(0) == 4) {
			mes "[Guia da Sala de Leilão]";
			mes "Otimo" Bem, então";
			mes "Espero que se divirta";
			mes "e aproveite o leilão~";
		}else{
			mes "[Guia da Sala de Leilão]";
			mes "Otimo! Bem, então";
			mes "aproveite o leilão.";
		}
		close2;
		switch(getarg(0)) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[Guia da Sala de Leilão]";
	mes "Tudo bem então,";
	mes "vejo você mais tarde.";
	mes "Se você mudar de";
	mes "ideia, por favor, venha";
	mes "aproveitar os leilões";
	close;
}
